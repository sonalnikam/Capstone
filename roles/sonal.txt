- name: Add a new user named devops
  user:
    name: devops
    shell: /bin/bash
    password: "devops"

- name: Add to sudoers
  copy:
    dest: "/etc/sudoers.d/devops"
    content: "devops  ALL=(ALL)  NOPASSWD: ALL"
    
- name: create ssh dir
  command: "{{ item }}"
  with_items:
   - mkdir /home/devops/.ssh
   - chmod 700 /home/devops/.ssh   
   - touch /home/devops/.ssh/authorized_keys
   - cat /home/ubuntu/.ssh/authorized_keys >> /home/devops/.ssh/authorized_keys
